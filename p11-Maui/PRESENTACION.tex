\begin{frame}
  \frametitle{Características de Maui I}
  \begin{enumerate}
  \item Múltiples políticas de planificación.
  \item Ajuste de prioridades de forma dinámica.
  \item Reserva de nodos.
  \item Sistema de control de procesos avanzado.
  \end{enumerate}
\end{frame}


\begin{frame}
  \frametitle{Características de Maui II}
  \begin{enumerate}
  \item \textbf{Control de Tráfico:} Gestiona las tareas para que no
    interfieran entre si. 
  \item \textbf{Centrado en Objetivos:} Configuración de cluster según la tarea.
  \item \textbf{Optimizaciones:} Supeditado a los dos puntos
    anteriores, el planificador intenta aprovechar los recursos del cluster.
  \end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{Job}
  
  \textbf{Job: } La unidad de trabajo de Maui (lo que va a ejecutar).\\
  Un \emph{job} está formado a su vez por:

  \begin{enumerate}
  \item Dueño del trabajo.
  \item Varios \textbf{Requisitos} (cada uno con sus limitaciones de recursos).
  \item Tiempo máximo de ejecución.
  \end{enumerate}
 
\end{frame}


\begin{frame}
  \frametitle{Requisitos}
  Un \textbf{Requisito} está formado por:
  \begin{itemize}
  \item Definición de la tarea.
  \item Límite de recursos: Procesadores, memoria, swap, disk.
  \item Task count: Nº de instancias de la tarea a ejecutar en el requisito.
  \item Task list: Lista de nodos en los que la tarea tiene que ser
    ejecutada.
  \item Recursos estadísticos: Qué aspectos de la tarea monitorear.
  \end{itemize}

  \begin{block}{Ejemplo}
    Un \textbf{trabajo} puede consistir en 2 requisitos:
    \begin{enumerate}
    \item 1 nodo, 512 MB RAM como mínimo, ejecutar la tarea X
    \item 24 nodos, 128 MB como mínimo, ejecutar la tarea Y
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Tipos de Nodos}
  Maui contempla los siguientes tipos de nodos:
  \begin{enumerate}
  \item \textbf{SHARED: } Se puede ejecutar cualquier combinación de tareas
    provenientes de cualquier job.
  \item \textbf{SINGLEUSER: } Se pueden ejecutar distintas tareas siempre
    que sean del mismo usuario.
  \item \textbf{SINGLEJOB: } Se pueden ejecutar distintas tareas siempre
    que sean del mismo job.
  \item \textbf{SINGLETASK: } Solamente se puede ejecutar una tarea simultáneamente.
  \end{enumerate}
\end{frame}



\begin{frame}
  \frametitle{Prioridad a las tareas}
  La prioridad se calcula de acuerdo a cada \alert{componente}:\\
  \begin{center}
    \texttt{Nodos}, \texttt{Procesadores}, \texttt{Memoria}, \texttt{Swap}, \texttt{Disco}
  \end{center}

  \begin{center}
    \texttt{<COMPONENT WEIGHT> * <SUBCOMPONENT WEIGHT> * <PRIORITY
      SUBCOMPONENT VALUE>}
  \end{center}

  \begin{enumerate}
  \item Gestión de trabajos mediante una FIFO de prioridades.
  \item Posibilidad de modificar las prioridades en tiempo real.
  \end{enumerate}
\end{frame}


\begin{frame}
  \frametitle{Asignación de nodos}
  Maui proporciona distintas políticas para la asignación de nodos:

  \begin{itemize}
  \item \textbf{CPULOAD: } Máximo número de cpus sin usar.
  \item \textbf{FIRSTAVAILABLE: } Primer nodo libre.
  \item \textbf{MINRESOURCE: } Nodo que desaproveche menos recursos.
  \item \textbf{CONTIGUOUS: } Asignar nodos contiguos.
  \item \textbf{MAXBALANCE: } Intenta buscar los mejores nodos para el
    trabajo (lento).
  \item \textbf{FASTEST: } Selección de los nodos más rápidos.
  \item \textbf{LOCAL: } Algoritmo propio de cada nodo (opcional).
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Justicia}
  \alert{Prioridad de un \emph{job}.} 

  \alert{Gestión de memoria:} Se tiene en cuenta el espacio disponible.

  \alert{FairShare:} Mecanismo de registro histórico de uso de los
  recursos. Se calcula el uso de recursos por parte de los diferentes
  autorizados \emph{(p.e., groups, user, QoS,...)}

  \alert{Directivas limitantes:} Opciones de configuración y permisos
  de un usuario.
  \begin{itemize}
  \item \textbf{MAXJOB:} máximo número de \emph{jobs} simultáneos.
  \item \textbf{MAXPROC:} máximo número de procesadores utilizados 
  \item \textbf{MAXWC:} máximo tiempo que un \emph{job} puede estar activo.
  \item \textbf{MAXNODE:} máximo número de nodos.
  \item \textbf{MAXMEM:} memoria máxima que puede almacenar entre
    todos los \emph{jobs}.
  \end{itemize}

\end{frame}
